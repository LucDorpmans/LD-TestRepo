Download My GitHub Scripts:

# Command to get my scripts from the GitHub:
$MyScript = "Get-MyScripts.ps1"
Invoke-Webrequest -Uri "https://raw.githubusercontent.com/LucDorpmans/Test-Repo/main/$MyScript"  -Outfile "$env:USERPROFILE\Downloads\$MyScript"
PSEdit  ("$env:USERPROFILE\Downloads\$MyScript")
